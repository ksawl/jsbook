```javascript
// script.js
class ImageGallery {
  constructor() {
    this.images = this.loadImages();
    this.currentImageIndex = 0;
    this.filteredImages = [...this.images];

    this.gallery = document.getElementById("gallery");
    this.imageModal = document.getElementById("image-modal");
    this.addModal = document.getElementById("add-modal");
    this.searchInput = document.getElementById("search-input");

    this.init();
  }

  init() {
    this.renderGallery();
    this.setupEventListeners();
  }

  setupEventListeners() {
    // Кнопка добавления изображения
    document.getElementById("add-image-btn").addEventListener("click", () => {
      this.showAddModal();
    });

    // Поиск
    this.searchInput.addEventListener("input", (e) => {
      this.filterImages(e.target.value);
    });

    // Форма добавления
    document
      .getElementById("add-image-form")
      .addEventListener("submit", (e) => {
        e.preventDefault();
        this.addImage();
      });

    // Закрытие модальных окон
    document.querySelectorAll(".close").forEach((closeBtn) => {
      closeBtn.addEventListener("click", (e) => {
        this.closeModal(e.target.closest(".modal"));
      });
    });

    // Навигация в модальном окне
    document.getElementById("prev-btn").addEventListener("click", () => {
      this.showPreviousImage();
    });

    document.getElementById("next-btn").addEventListener("click", () => {
      this.showNextImage();
    });

    // Закрытие модального окна по клику вне содержимого
    window.addEventListener("click", (e) => {
      if (e.target.classList.contains("modal")) {
        this.closeModal(e.target);
      }
    });

    // Клавиатурная навигация
    document.addEventListener("keydown", (e) => {
      if (this.imageModal.style.display === "block") {
        if (e.key === "ArrowLeft") this.showPreviousImage();
        if (e.key === "ArrowRight") this.showNextImage();
        if (e.key === "Escape") this.closeModal(this.imageModal);
      }
    });
  }

  renderGallery() {
    this.gallery.innerHTML = "";

    if (this.filteredImages.length === 0) {
      this.gallery.innerHTML = "<p>Изображения не найдены</p>";
      return;
    }

    this.filteredImages.forEach((image, index) => {
      const item = this.createGalleryItem(image, index);
      this.gallery.appendChild(item);
    });
  }

  createGalleryItem(image, index) {
    const item = document.createElement("div");
    item.className = "gallery-item";
    item.style.position = "relative";

    item.innerHTML = `
            <img src="${image.url}" alt="${image.title}" loading="lazy">
            <div class="gallery-item-info">
                <h3>${image.title}</h3>
                <p>${image.description}</p>
            </div>
            <button class="delete-btn" data-index="${image.id}">×</button>
        `;

    // Открытие модального окна по клику на изображение
    item.addEventListener("click", (e) => {
      if (!e.target.classList.contains("delete-btn")) {
        this.showImageModal(index);
      }
    });

    // Удаление изображения
    const deleteBtn = item.querySelector(".delete-btn");
    deleteBtn.addEventListener("click", (e) => {
      e.stopPropagation();
      this.deleteImage(image.id);
    });

    return item;
  }

  showImageModal(index) {
    this.currentImageIndex = index;
    const image = this.filteredImages[index];

    document.getElementById("modal-image").src = image.url;
    document.getElementById("modal-title").textContent = image.title;
    document.getElementById("modal-description").textContent =
      image.description;

    // Обновление навигации
    document.getElementById("prev-btn").disabled = index === 0;
    document.getElementById("next-btn").disabled =
      index === this.filteredImages.length - 1;

    this.imageModal.style.display = "block";
    document.body.style.overflow = "hidden";
  }

  showPreviousImage() {
    if (this.currentImageIndex > 0) {
      this.showImageModal(this.currentImageIndex - 1);
    }
  }

  showNextImage() {
    if (this.currentImageIndex < this.filteredImages.length - 1) {
      this.showImageModal(this.currentImageIndex + 1);
    }
  }

  showAddModal() {
    this.addModal.style.display = "block";
    document.body.style.overflow = "hidden";
    document.getElementById("image-url").focus();
  }

  closeModal(modal) {
    modal.style.display = "none";
    document.body.style.overflow = "auto";

    // Очистка формы при закрытии модального окна добавления
    if (modal === this.addModal) {
      document.getElementById("add-image-form").reset();
    }
  }

  addImage() {
    const url = document.getElementById("image-url").value;
    const title = document.getElementById("image-title").value;
    const description = document.getElementById("image-description").value;

    // Проверка URL изображения
    if (!this.isValidImageUrl(url)) {
      alert("Пожалуйста, введите корректный URL изображения");
      return;
    }

    const newImage = {
      id: Date.now(),
      url: url,
      title: title,
      description: description,
      dateAdded: new Date(),
    };

    this.images.unshift(newImage);
    this.saveImages();
    this.filterImages(this.searchInput.value);
    this.closeModal(this.addModal);

    // Показать уведомление
    this.showNotification("Изображение добавлено!");
  }

  deleteImage(id) {
    if (confirm("Удалить это изображение?")) {
      this.images = this.images.filter((img) => img.id !== id);
      this.saveImages();
      this.filterImages(this.searchInput.value);
      this.showNotification("Изображение удалено");
    }
  }

  filterImages(query) {
    const lowerQuery = query.toLowerCase();
    this.filteredImages = this.images.filter(
      (image) =>
        image.title.toLowerCase().includes(lowerQuery) ||
        image.description.toLowerCase().includes(lowerQuery)
    );
    this.renderGallery();
  }

  isValidImageUrl(url) {
    try {
      new URL(url);
      return /\.(jpg|jpeg|png|gif|webp|svg)$/i.test(url);
    } catch {
      return false;
    }
  }

  showNotification(message) {
    // Создание уведомления
    const notification = document.createElement("div");
    notification.className = "notification";
    notification.textContent = message;
    notification.style.cssText = `
            position: fixed;
            top: 20px;
            right: 20px;
            background: #27ae60;
            color: white;
            padding: 15px 20px;
            border-radius: 5px;
            z-index: 10000;
            transform: translateX(100%);
            transition: transform 0.3s ease;
        `;

    document.body.appendChild(notification);

    // Анимация появления
    setTimeout(() => {
      notification.style.transform = "translateX(0)";
    }, 100);

    // Автоматическое удаление
    setTimeout(() => {
      notification.style.transform = "translateX(100%)";
      setTimeout(() => {
        if (notification.parentNode) {
          notification.parentNode.removeChild(notification);
        }
      }, 300);
    }, 3000);
  }

  loadImages() {
    const saved = localStorage.getItem("gallery-images");
    if (saved) {
      return JSON.parse(saved);
    }

    // Демо-данные
    return [
      {
        id: 1,
        url: "https://picsum.photos/400/300?random=1",
        title: "Природа",
        description: "Красивый пейзаж с горами",
        dateAdded: new Date(),
      },
      {
        id: 2,
        url: "https://picsum.photos/400/300?random=2",
        title: "Архитектура",
        description: "Современное здание",
        dateAdded: new Date(),
      },
      {
        id: 3,
        url: "https://picsum.photos/400/300?random=3",
        title: "Технологии",
        description: "Инновационные решения",
        dateAdded: new Date(),
      },
    ];
  }

  saveImages() {
    localStorage.setItem("gallery-images", JSON.stringify(this.images));
  }
}

// Инициализация галереи
document.addEventListener("DOMContentLoaded", () => {
  new ImageGallery();
});
```

### JavaScript для списка покупок

```javascript
// shopping-script.js
class ShoppingList {
  constructor() {
    this.items = this.loadItems();
    this.currentFilter = "all";
    this.init();
  }

  init() {
    this.renderList();
    this.updateStats();
    this.setupEventListeners();
  }

  setupEventListeners() {
    // Добавление товара
    document.getElementById("add-btn").addEventListener("click", () => {
      this.addItem();
    });

    // Enter в поле ввода
    document.getElementById("item-name").addEventListener("keypress", (e) => {
      if (e.key === "Enter") {
        this.addItem();
      }
    });

    // Фильтры
    document.querySelectorAll(".filter-btn").forEach((btn) => {
      btn.addEventListener("click", (e) => {
        this.setFilter(e.target.dataset.filter);
        this.updateActiveFilter(e.target);
      });
    });

    // Очистка
    document.getElementById("clear-completed").addEventListener("click", () => {
      this.clearCompleted();
    });

    document.getElementById("clear-all").addEventListener("click", () => {
      this.clearAll();
    });
  }

  addItem() {
    const nameInput = document.getElementById("item-name");
    const categorySelect = document.getElementById("item-category");
    const quantityInput = document.getElementById("item-quantity");

    const name = nameInput.value.trim();
    if (!name) {
      nameInput.focus();
      return;
    }

    const item = {
      id: Date.now(),
      name: name,
      category: categorySelect.value,
      quantity: parseInt(quantityInput.value),
      completed: false,
      addedDate: new Date(),
    };

    this.items.unshift(item);
    this.saveItems();
    this.renderList();
    this.updateStats();

    // Очистка формы
    nameInput.value = "";
    quantityInput.value = 1;
    nameInput.focus();
  }

  toggleItem(id) {
    const item = this.items.find((item) => item.id === id);
    if (item) {
      item.completed = !item.completed;
      this.saveItems();
      this.renderList();
      this.updateStats();
    }
  }

  deleteItem(id) {
    this.items = this.items.filter((item) => item.id !== id);
    this.saveItems();
    this.renderList();
    this.updateStats();
  }

  editItem(id, newName) {
    const item = this.items.find((item) => item.id === id);
    if (item && newName.trim()) {
      item.name = newName.trim();
      this.saveItems();
      this.renderList();
    }
  }

  updateQuantity(id, newQuantity) {
    const item = this.items.find((item) => item.id === id);
    if (item && newQuantity > 0) {
      item.quantity = newQuantity;
      this.saveItems();
      this.renderList();
    }
  }

  renderList() {
    const list = document.getElementById("shopping-list");
    const filteredItems = this.getFilteredItems();

    if (filteredItems.length === 0) {
      list.innerHTML = '<div class="empty-state">Список пуст</div>';
      return;
    }

    list.innerHTML = "";
    filteredItems.forEach((item) => {
      const itemElement = this.createItemElement(item);
      list.appendChild(itemElement);
    });
  }

  createItemElement(item) {
    const itemDiv = document.createElement("div");
    itemDiv.className = `shopping-item ${item.completed ? "completed" : ""}`;
    itemDiv.dataset.id = item.id;

    itemDiv.innerHTML = `
            <div class="item-checkbox">
                <input type="checkbox" ${item.completed ? "checked" : ""} 
                       onchange="shoppingList.toggleItem(${item.id})">
            </div>
            <div class="item-content">
                <div class="item-name" ondblclick="shoppingList.startEdit(${
                  item.id
                })">${item.name}</div>
                <div class="item-meta">
                    <span class="category">${this.getCategoryIcon(
                      item.category
                    )} ${this.getCategoryName(item.category)}</span>
                    <span class="quantity">Количество: ${item.quantity}</span>
                </div>
            </div>
            <div class="item-controls">
                <button onclick="shoppingList.startEdit(${item.id})">✏️</button>
                <input type="number" value="${item.quantity}" min="1" 
                       onchange="shoppingList.updateQuantity(${
                         item.id
                       }, this.value)"
                       onclick="event.stopPropagation()">
                <button onclick="shoppingList.deleteItem(${
                  item.id
                })">🗑️</button>
            </div>
        `;

    return itemDiv;
  }

  startEdit(id) {
    const itemElement = document.querySelector(`[data-id="${id}"]`);
    const nameElement = itemElement.querySelector(".item-name");
    const currentName = nameElement.textContent;

    const input = document.createElement("input");
    input.type = "text";
    input.value = currentName;
    input.className = "edit-input";

    const saveEdit = () => {
      this.editItem(id, input.value);
    };

    const cancelEdit = () => {
      nameElement.textContent = currentName;
      nameElement.style.display = "block";
      input.remove();
    };

    input.addEventListener("keypress", (e) => {
      if (e.key === "Enter") saveEdit();
      if (e.key === "Escape") cancelEdit();
    });

    input.addEventListener("blur", saveEdit);

    nameElement.style.display = "none";
    nameElement.parentNode.insertBefore(input, nameElement);
    input.focus();
    input.select();
  }

  getFilteredItems() {
    switch (this.currentFilter) {
      case "active":
        return this.items.filter((item) => !item.completed);
      case "completed":
        return this.items.filter((item) => item.completed);
      case "all":
        return this.items;
      default:
        return this.items.filter(
          (item) => item.category === this.currentFilter
        );
    }
  }

  setFilter(filter) {
    this.currentFilter = filter;
    this.renderList();
  }

  updateActiveFilter(activeBtn) {
    document.querySelectorAll(".filter-btn").forEach((btn) => {
      btn.classList.remove("active");
    });
    activeBtn.classList.add("active");
  }

  updateStats() {
    const total = this.items.length;
    const completed = this.items.filter((item) => item.completed).length;
    const remaining = total - completed;

    document.getElementById("total-count").textContent = total;
    document.getElementById("bought-count").textContent = completed;
    document.getElementById("remaining-count").textContent = remaining;
  }

  clearCompleted() {
    if (confirm("Удалить все купленные товары?")) {
      this.items = this.items.filter((item) => !item.completed);
      this.saveItems();
      this.renderList();
      this.updateStats();
    }
  }

  clearAll() {
    if (confirm("Очистить весь список?")) {
      this.items = [];
      this.saveItems();
      this.renderList();
      this.updateStats();
    }
  }

  getCategoryIcon(category) {
    const icons = {
      food: "🍎",
      household: "🧽",
      clothes: "👕",
      electronics: "📱",
      other: "📦",
    };
    return icons[category] || "📦";
  }

  getCategoryName(category) {
    const names = {
      food: "Продукты",
      household: "Хозтовары",
      clothes: "Одежда",
      electronics: "Электроника",
      other: "Другое",
    };
    return names[category] || "Другое";
  }

  loadItems() {
    const saved = localStorage.getItem("shopping-list");
    return saved ? JSON.parse(saved) : [];
  }

  saveItems() {
    localStorage.setItem("shopping-list", JSON.stringify(this.items));
  }
}
```

## Проект 3: Квиз-приложение

Создайте интерактивную викторину с вопросами и подсчетом результатов.

### Основной функционал

```javascript
class QuizApp {
  constructor() {
    this.questions = this.loadQuestions();
    this.currentQuestionIndex = 0;
    this.score = 0;
    this.userAnswers = [];
    this.timeLeft = 60; // секунды
    this.timer = null;

    this.init();
  }

  loadQuestions() {
    return [
      {
        question: "Что означает DOM?",
        options: [
          "Document Object Model",
          "Data Object Management",
          "Dynamic Object Method",
          "Document Operation Mode",
        ],
        correct: 0,
      },
      {
        question: "Какой метод используется для поиска элемента по ID?",
        options: [
          "querySelector()",
          "getElementById()",
          "findElementById()",
          "getElement()",
        ],
        correct: 1,
      },
      {
        question: "Что возвращает querySelectorAll()?",
        options: ["Array", "HTMLCollection", "NodeList", "Object"],
        correct: 2,
      },
    ];
  }

  init() {
    this.createQuizInterface();
    this.showQuestion();
    this.startTimer();
  }

  createQuizInterface() {
    document.body.innerHTML = `
            <div class="quiz-container">
                <header class="quiz-header">
                    <h1>JavaScript Квиз</h1>
                    <div class="quiz-info">
                        <span>Вопрос <span id="current-q">1</span> из <span id="total-q">${this.questions.length}</span></span>
                        <span>Время: <span id="timer">${this.timeLeft}</span>с</span>
                    </div>
                </header>
                
                <div class="progress-bar">
                    <div class="progress" id="progress"></div>
                </div>
                
                <div class="question-container" id="question-container">
                    <!-- Вопрос будет добавлен динамически -->
                </div>
                
                <div class="quiz-controls">
                    <button id="prev-btn" disabled>Предыдущий</button>
                    <button id="next-btn" disabled>Следующий</button>
                    <button id="finish-btn" style="display: none;">Завершить</button>
                </div>
            </div>
        `;

    this.setupEventListeners();
  }

  setupEventListeners() {
    document.getElementById("prev-btn").addEventListener("click", () => {
      this.previousQuestion();
    });

    document.getElementById("next-btn").addEventListener("click", () => {
      this.nextQuestion();
    });

    document.getElementById("finish-btn").addEventListener("click", () => {
      this.finishQuiz();
    });
  }

  showQuestion() {
    const question = this.questions[this.currentQuestionIndex];
    const container = document.getElementById("question-container");

    container.innerHTML = `
            <div class="question">
                <h2>${question.question}</h2>
                <div class="options">
                    ${question.options
                      .map(
                        (option, index) => `
                        <label class="option">
                            <input type="radio" name="answer" value="${index}" 
                                   ${
                                     this.userAnswers[
                                       this.currentQuestionIndex
                                     ] === index
                                       ? "checked"
                                       : ""
                                   }>
                            <span class="option-text">${option}</span>
                        </label>
                    `
                      )
                      .join("")}
                </div>
            </div>
        `;

    // Обновление интерфейса
    document.getElementById("current-q").textContent =
      this.currentQuestionIndex + 1;
    this.updateProgress();
    this.updateButtons();

    // Обработка выбора ответа
    const options = container.querySelectorAll('input[name="answer"]');
    options.forEach((option) => {
      option.addEventListener("change", (e) => {
        this.selectAnswer(parseInt(e.target.value));
      });
    });
  }

  selectAnswer(answerIndex) {
    this.userAnswers[this.currentQuestionIndex] = answerIndex;
    this.updateButtons();
  }

  updateButtons() {
    const prevBtn = document.getElementById("prev-btn");
    const nextBtn = document.getElementById("next-btn");
    const finishBtn = document.getElementById("finish-btn");

    prevBtn.disabled = this.currentQuestionIndex === 0;

    const hasAnswer = this.userAnswers[this.currentQuestionIndex] !== undefined;
    const isLastQuestion =
      this.currentQuestionIndex === this.questions.length - 1;

    if (isLastQuestion) {
      nextBtn.style.display = "none";
      finishBtn.style.display = hasAnswer ? "block" : "none";
    } else {
      nextBtn.style.display = "block";
      nextBtn.disabled = !hasAnswer;
      finishBtn.style.display = "none";
    }
  }

  updateProgress() {
    const progress =
      ((this.currentQuestionIndex + 1) / this.questions.length) * 100;
    document.getElementById("progress").style.width = progress + "%";
  }

  nextQuestion() {
    if (this.currentQuestionIndex < this.questions.length - 1) {
      this.currentQuestionIndex++;
      this.showQuestion();
    }
  }

  previousQuestion() {
    if (this.currentQuestionIndex > 0) {
      this.currentQuestionIndex--;
      this.showQuestion();
    }
  }

  startTimer() {
    this.timer = setInterval(() => {
      this.timeLeft--;
      document.getElementById("timer").textContent = this.timeLeft;

      if (this.timeLeft <= 0) {
        this.finishQuiz();
      }
    }, 1000);
  }

  finishQuiz() {
    clearInterval(this.timer);
    this.calculateScore();
    this.showResults();
  }

  calculateScore() {
    this.score = 0;
    this.questions.forEach((question, index) => {
      if (this.userAnswers[index] === question.correct) {
        this.score++;
      }
    });
  }

  showResults() {
    const percentage = Math.round((this.score / this.questions.length) * 100);

    document.body.innerHTML = `
            <div class="results-container">
                <h1>Результаты квиза</h1>
                <div class="score-display">
                    <div class="score-circle">
                        <span class="score-number">${percentage}%</span>
                    </div>
                    <p>Правильных ответов: ${this.score} из ${
      this.questions.length
    }</p>
                </div>
                
                <div class="detailed-results">
                    <h3>Детальные результаты:</h3>
                    ${this.questions
                      .map((question, index) => {
                        const userAnswer = this.userAnswers[index];
                        const isCorrect = userAnswer === question.correct;

                        return `
                            <div class="result-item ${
                              isCorrect ? "correct" : "incorrect"
                            }">
                                <div class="question-text">${
                                  question.question
                                }</div>
                                <div class="answer-info">
                                    <div>Ваш ответ: ${
                                      userAnswer !== undefined
                                        ? question.options[userAnswer]
                                        : "Не отвечен"
                                    }</div>
                                    <div>Правильный ответ: ${
                                      question.options[question.correct]
                                    }</div>
                                </div>
                            </div>
                        `;
                      })
                      .join("")}
                </div>
                
                <button onclick="location.reload()">Пройти снова</button>
            </div>
        `;
  }
}

// Инициализация квиза
document.addEventListener("DOMContentLoaded", () => {
  new QuizApp();
});
```
