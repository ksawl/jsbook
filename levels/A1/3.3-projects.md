# **3.3. Проекты для закрепления**

## Проект 1: Калькулятор функций

Создайте набор функций для выполнения математических операций.

### Задача

Написать функции для основных математических операций и продемонстрировать их использование.

### Требования

1. Создать функции для сложения, вычитания, умножения и деления
2. Добавить функцию для возведения в степень
3. Создать функцию-калькулятор, которая принимает операцию и два числа

### Решение

```javascript
// Основные математические операции
function add(a, b) {
  return a + b;
}

function subtract(a, b) {
  return a - b;
}

function multiply(a, b) {
  return a * b;
}

function divide(a, b) {
  if (b === 0) {
    return "Ошибка: деление на ноль";
  }
  return a / b;
}

function power(base, exponent) {
  return Math.pow(base, exponent);
}

// Функция-калькулятор
function calculator(operation, a, b) {
  switch (operation) {
    case "+":
      return add(a, b);
    case "-":
      return subtract(a, b);
    case "*":
      return multiply(a, b);
    case "/":
      return divide(a, b);
    case "**":
      return power(a, b);
    default:
      return "Неизвестная операция";
  }
}

// Демонстрация использования
console.log("Сложение:", calculator("+", 10, 5));
console.log("Вычитание:", calculator("-", 10, 5));
console.log("Умножение:", calculator("*", 10, 5));
console.log("Деление:", calculator("/", 10, 5));
console.log("Возведение в степень:", calculator("**", 2, 3));
```

## Проект 2: Работа с массивами студентов

Создайте систему для управления списком студентов.

### Задача

Разработать функции для работы со списком студентов и их оценками.

### Требования

1. Создать массив студентов с их оценками
2. Написать функции для добавления студентов
3. Создать функцию для подсчета средней оценки
4. Добавить функцию поиска студента

### Решение

```javascript
// Массив студентов
let students = [
  { name: "Иван", grades: [5, 4, 5, 3, 4] },
  { name: "Мария", grades: [5, 5, 4, 5, 5] },
  { name: "Петр", grades: [3, 4, 3, 4, 3] },
];

// Добавление нового студента
function addStudent(name, grades = []) {
  students.push({
    name: name,
    grades: grades,
  });
  console.log(`Студент ${name} добавлен в список`);
}

// Вычисление средней оценки студента
function calculateAverage(grades) {
  if (grades.length === 0) return 0;

  const sum = grades.reduce((total, grade) => total + grade, 0);
  return sum / grades.length;
}

// Получение информации о студенте
function getStudentInfo(name) {
  const student = students.find((s) => s.name === name);

  if (!student) {
    return `Студент ${name} не найден`;
  }

  const average = calculateAverage(student.grades);
  return {
    name: student.name,
    grades: student.grades,
    average: average.toFixed(2),
  };
}

// Получение лучших студентов
function getTopStudents(minAverage = 4.0) {
  return students
    .map((student) => ({
      ...student,
      average: calculateAverage(student.grades),
    }))
    .filter((student) => student.average >= minAverage)
    .sort((a, b) => b.average - a.average);
}

// Демонстрация использования
console.log("Все студенты:", students);

addStudent("Анна", [5, 5, 4, 5]);

console.log("Информация о Марии:", getStudentInfo("Мария"));
console.log("Лучшие студенты:", getTopStudents(4.0));
```

## Проект 3: Генератор паролей

Создайте функцию для генерации безопасных паролей.

### Задача

Написать функцию, которая генерирует случайные пароли с заданными параметрами.

### Требования

1. Возможность задать длину пароля
2. Выбор типов символов (буквы, цифры, специальные символы)
3. Проверка силы пароля

### Решение

```javascript
// Наборы символов
const charSets = {
  lowercase: "abcdefghijklmnopqrstuvwxyz",
  uppercase: "ABCDEFGHIJKLMNOPQRSTUVWXYZ",
  numbers: "0123456789",
  special: "!@#$%^&*()_+-=[]{}|;:,.<>?",
};

// Генерация случайного пароля
function generatePassword(length = 12, options = {}) {
  const {
    includeLowercase = true,
    includeUppercase = true,
    includeNumbers = true,
    includeSpecial = false,
  } = options;

  let availableChars = "";

  if (includeLowercase) availableChars += charSets.lowercase;
  if (includeUppercase) availableChars += charSets.uppercase;
  if (includeNumbers) availableChars += charSets.numbers;
  if (includeSpecial) availableChars += charSets.special;

  if (availableChars === "") {
    return "Ошибка: не выбран ни один тип символов";
  }

  let password = "";
  for (let i = 0; i < length; i++) {
    const randomIndex = Math.floor(Math.random() * availableChars.length);
    password += availableChars[randomIndex];
  }

  return password;
}

// Проверка силы пароля
function checkPasswordStrength(password) {
  let score = 0;
  let feedback = [];

  // Проверка длины
  if (password.length >= 8) {
    score += 1;
  } else {
    feedback.push("Пароль должен содержать минимум 8 символов");
  }

  // Проверка на строчные буквы
  if (/[a-z]/.test(password)) {
    score += 1;
  } else {
    feedback.push("Добавьте строчные буквы");
  }

  // Проверка на заглавные буквы
  if (/[A-Z]/.test(password)) {
    score += 1;
  } else {
    feedback.push("Добавьте заглавные буквы");
  }

  // Проверка на цифры
  if (/\d/.test(password)) {
    score += 1;
  } else {
    feedback.push("Добавьте цифры");
  }

  // Проверка на специальные символы
  if (/[!@#$%^&*()_+\-=\[\]{}|;:,.<>?]/.test(password)) {
    score += 1;
  } else {
    feedback.push("Добавьте специальные символы");
  }

  // Определение силы пароля
  let strength;
  if (score <= 2) strength = "Слабый";
  else if (score <= 3) strength = "Средний";
  else if (score <= 4) strength = "Сильный";
  else strength = "Очень сильный";

  return {
    strength,
    score,
    feedback,
  };
}

// Демонстрация использования
const password1 = generatePassword(12);
console.log("Сгенерированный пароль:", password1);
console.log("Сила пароля:", checkPasswordStrength(password1));

const password2 = generatePassword(16, {
  includeLowercase: true,
  includeUppercase: true,
  includeNumbers: true,
  includeSpecial: true,
});
console.log("Сложный пароль:", password2);
console.log("Сила сложного пароля:", checkPasswordStrength(password2));
```
