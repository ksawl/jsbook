# **6.2. Введение в DOM**

## Что такое DOM?

**DOM** (Document Object Model) — это программный интерфейс для HTML и XML документов. Он представляет структуру документа в виде дерева объектов, которые можно изменять с помощью JavaScript.

### Основные концепции DOM

- **Документ** — корневой объект, представляющий всю веб-страницу
- **Элементы** — HTML-теги, представленные как объекты
- **Узлы** — все части документа (элементы, текст, атрибуты)
- **Дерево** — иерархическая структура узлов

## Структура DOM-дерева

### HTML документ

```html
<!DOCTYPE html>
<html lang="ru">
  <head>
    <meta charset="UTF-8" />
    <title>Моя страница</title>
  </head>
  <body>
    <header>
      <h1>Заголовок сайта</h1>
      <nav>
        <ul>
          <li><a href="#home">Главная</a></li>
          <li><a href="#about">О нас</a></li>
        </ul>
      </nav>
    </header>
    <main>
      <section>
        <h2>Содержание</h2>
        <p>Текст параграфа</p>
      </section>
    </main>
  </body>
</html>
```

### DOM-дерево

```
Document
└── html
    ├── head
    │   ├── meta
    │   └── title
    │       └── "Моя страница" (текстовый узел)
    └── body
        ├── header
        │   ├── h1
        │   │   └── "Заголовок сайта" (текстовый узел)
        │   └── nav
        │       └── ul
        │           ├── li
        │           │   └── a
        │           │       └── "Главная" (текстовый узел)
        │           └── li
        │               └── a
        │                   └── "О нас" (текстовый узел)
        └── main
            └── section
                ├── h2
                │   └── "Содержание" (текстовый узел)
                └── p
                    └── "Текст параграфа" (текстовый узел)
```

## Типы узлов в DOM

### Основные типы узлов

1. **Element Node** (Node.ELEMENT_NODE = 1) — HTML-элементы
2. **Text Node** (Node.TEXT_NODE = 3) — текстовое содержимое
3. **Comment Node** (Node.COMMENT_NODE = 8) — комментарии
4. **Document Node** (Node.DOCUMENT_NODE = 9) — корневой документ

```javascript
// Проверка типа узла
const element = document.querySelector("h1");
console.log(element.nodeType); // 1 (ELEMENT_NODE)
console.log(element.nodeName); // "H1"

const textNode = element.firstChild;
console.log(textNode.nodeType); // 3 (TEXT_NODE)
console.log(textNode.nodeName); // "#text"
```

## Объект `document`

Объект `document` представляет всю HTML-страницу и предоставляет методы для работы с DOM.

### Свойства document

```javascript
// Информация о документе
console.log(document.title); // Заголовок страницы
console.log(document.URL); // URL страницы
console.log(document.domain); // Домен
console.log(document.referrer); // Страница, с которой пришли
console.log(document.lastModified); // Дата последнего изменения

// Корневые элементы
console.log(document.documentElement); // <html>
console.log(document.head); // <head>
console.log(document.body); // <body>

// Коллекции элементов
console.log(document.images); // Все <img>
console.log(document.links); // Все <a> с href
console.log(document.forms); // Все <form>
console.log(document.scripts); // Все <script>
```

### Состояние загрузки документа

```javascript
// Проверка состояния загрузки
console.log(document.readyState);
// "loading" - документ загружается
// "interactive" - загрузка завершена, но ресурсы могут грузиться
// "complete" - все ресурсы загружены

// Событие готовности DOM
document.addEventListener("DOMContentLoaded", function () {
  console.log("DOM полностью загружен");
});

// Событие полной загрузки страницы
window.addEventListener("load", function () {
  console.log("Страница полностью загружена");
});
```

## Навигация по DOM

### Родительские элементы

```javascript
const element = document.querySelector(".child");

// Родительский элемент
console.log(element.parentNode); // Любой родительский узел
console.log(element.parentElement); // Родительский элемент (только Element)

// Корневые элементы
console.log(element.ownerDocument); // document
console.log(element.getRootNode()); // корневой узел
```

### Дочерние элементы

```javascript
const parent = document.querySelector(".parent");

// Все дочерние узлы (включая текстовые)
console.log(parent.childNodes); // NodeList всех детей
console.log(parent.firstChild); // Первый дочерний узел
console.log(parent.lastChild); // Последний дочерний узел

// Только элементы (без текстовых узлов)
console.log(parent.children); // HTMLCollection только элементов
console.log(parent.firstElementChild); // Первый дочерний элемент
console.log(parent.lastElementChild); // Последний дочерний элемент
console.log(parent.childElementCount); // Количество дочерних элементов
```

### Соседние элементы

```javascript
const element = document.querySelector(".current");

// Все соседние узлы
console.log(element.nextSibling); // Следующий узел
console.log(element.previousSibling); // Предыдущий узел

// Только соседние элементы
console.log(element.nextElementSibling); // Следующий элемент
console.log(element.previousElementSibling); // Предыдущий элемент
```

## Практические примеры навигации

### Обход всех дочерних элементов

```javascript
function walkChildren(element) {
  console.log("Элемент:", element.tagName);

  // Перебор всех дочерних элементов
  for (let child of element.children) {
    walkChildren(child); // Рекурсивный обход
  }
}

// Обход всей страницы
walkChildren(document.body);
```

### Поиск родителя с определенным классом

```javascript
function findParentByClass(element, className) {
  let current = element.parentElement;

  while (current) {
    if (current.classList.contains(className)) {
      return current;
    }
    current = current.parentElement;
  }

  return null;
}

// Использование
const button = document.querySelector(".submit-btn");
const form = findParentByClass(button, "form-container");
console.log(form);
```

### Получение всех соседних элементов

```javascript
function getAllSiblings(element) {
  const siblings = [];
  let sibling = element.parentElement.firstElementChild;

  while (sibling) {
    if (sibling !== element) {
      siblings.push(sibling);
    }
    sibling = sibling.nextElementSibling;
  }

  return siblings;
}

// Использование
const currentElement = document.querySelector(".active");
const siblings = getAllSiblings(currentElement);
console.log("Соседние элементы:", siblings);
```

## Коллекции в DOM

### NodeList

NodeList — это коллекция узлов DOM:

```javascript
// Получение NodeList
const paragraphs = document.querySelectorAll("p");

console.log(paragraphs.length); // Количество элементов
console.log(paragraphs[0]); // Первый элемент
console.log(paragraphs.item(1)); // Второй элемент

// Итерация по NodeList
paragraphs.forEach((paragraph, index) => {
  console.log(`Параграф ${index}:`, paragraph.textContent);
});

// Преобразование в массив
const paragraphsArray = Array.from(paragraphs);
const paragraphsSpread = [...paragraphs];
```

### HTMLCollection

HTMLCollection — это живая коллекция HTML-элементов:

```javascript
// Получение HTMLCollection
const divs = document.getElementsByTagName("div");

console.log(divs.length); // Количество div-ов
console.log(divs[0]); // Первый div
console.log(divs.item(1)); // Второй div
console.log(divs.namedItem("myDiv")); // div с id="myDiv"

// HTMLCollection обновляется автоматически
console.log("До:", divs.length);
document.body.appendChild(document.createElement("div"));
console.log("После:", divs.length); // Увеличилось на 1

// Итерация (нет forEach!)
for (let i = 0; i < divs.length; i++) {
  console.log(divs[i]);
}

// Или преобразование в массив
Array.from(divs).forEach((div) => {
  console.log(div);
});
```

### Разница между NodeList и HTMLCollection

```javascript
// NodeList (статический снимок)
const nodeList = document.querySelectorAll("p");

// HTMLCollection (живая коллекция)
const htmlCollection = document.getElementsByTagName("p");

console.log("NodeList:", nodeList.length);
console.log("HTMLCollection:", htmlCollection.length);

// Добавляем новый параграф
const newParagraph = document.createElement("p");
document.body.appendChild(newParagraph);

console.log("NodeList после добавления:", nodeList.length); // Не изменилось
console.log("HTMLCollection после добавления:", htmlCollection.length); // Увеличилось
```

## Информация об элементах

### Свойства элементов

```javascript
const element = document.querySelector("h1");

// Основная информация
console.log(element.tagName); // "H1"
console.log(element.nodeName); // "H1"
console.log(element.nodeType); // 1
console.log(element.localName); // "h1"

// Содержимое
console.log(element.innerHTML); // HTML-содержимое
console.log(element.outerHTML); // HTML элемента с ним самим
console.log(element.textContent); // Весь текст
console.log(element.innerText); // Видимый текст

// Атрибуты
console.log(element.id); // id атрибут
console.log(element.className); // class атрибут
console.log(element.classList); // DOMTokenList классов
```

### Размеры и позиция

```javascript
const element = document.querySelector(".box");

// Размеры с границами и прокруткой
console.log(element.offsetWidth); // Ширина включая границы
console.log(element.offsetHeight); // Высота включая границы
console.log(element.offsetLeft); // Отступ слева от offsetParent
console.log(element.offsetTop); // Отступ сверху от offsetParent
console.log(element.offsetParent); // Родитель для позиционирования

// Размеры содержимого
console.log(element.clientWidth); // Ширина без границ
console.log(element.clientHeight); // Высота без границ
console.log(element.clientLeft); // Ширина левой границы
console.log(element.clientTop); // Ширина верхней границы

// Размеры с прокруткой
console.log(element.scrollWidth); // Полная ширина содержимого
console.log(element.scrollHeight); // Полная высота содержимого
console.log(element.scrollLeft); // Горизонтальная прокрутка
console.log(element.scrollTop); // Вертикальная прокрутка
```

## Практический пример: Анализатор DOM

```javascript
class DOMAnalyzer {
  constructor() {
    this.stats = {
      totalElements: 0,
      elementTypes: {},
      maxDepth: 0,
      elementsWithId: 0,
      elementsWithClass: 0,
    };
  }

  analyze(element = document.body, depth = 0) {
    this.stats.totalElements++;
    this.stats.maxDepth = Math.max(this.stats.maxDepth, depth);

    // Подсчет типов элементов
    const tagName = element.tagName.toLowerCase();
    this.stats.elementTypes[tagName] =
      (this.stats.elementTypes[tagName] || 0) + 1;

    // Подсчет элементов с id и классами
    if (element.id) this.stats.elementsWithId++;
    if (element.className) this.stats.elementsWithClass++;

    // Рекурсивный анализ дочерних элементов
    for (let child of element.children) {
      this.analyze(child, depth + 1);
    }

    return this.stats;
  }

  getReport() {
    const stats = this.analyze();

    console.group("🔍 Анализ DOM структуры");
    console.log("Всего элементов:", stats.totalElements);
    console.log("Максимальная вложенность:", stats.maxDepth);
    console.log("Элементов с ID:", stats.elementsWithId);
    console.log("Элементов с классами:", stats.elementsWithClass);

    console.log("Распределение по типам элементов:");
    console.table(stats.elementTypes);

    console.groupEnd();

    return stats;
  }

  findDeepestElement(
    element = document.body,
    depth = 0,
    deepest = { element: null, depth: -1 }
  ) {
    if (depth > deepest.depth) {
      deepest.element = element;
      deepest.depth = depth;
    }

    for (let child of element.children) {
      this.findDeepestElement(child, depth + 1, deepest);
    }

    return deepest;
  }

  findElementsWithoutChildren() {
    const leafElements = [];

    function traverse(element) {
      if (element.children.length === 0) {
        leafElements.push(element);
      } else {
        for (let child of element.children) {
          traverse(child);
        }
      }
    }

    traverse(document.body);
    return leafElements;
  }
}

// Использование анализатора
const analyzer = new DOMAnalyzer();
const report = analyzer.getReport();

const deepest = analyzer.findDeepestElement();
console.log("Самый глубокий элемент:", deepest);

const leafElements = analyzer.findElementsWithoutChildren();
console.log("Элементы без дочерних:", leafElements.length);
```
