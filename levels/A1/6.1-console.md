# **6.1. Работа с консолью**

## Глобальные объекты в браузере и Node.js

### Объект `window` в браузере

В браузере глобальный объект называется `window` и содержит все глобальные переменные и функции:

```javascript
// В браузере
console.log(window); // объект window
console.log(this === window); // true (в глобальной области видимости)

// Добавление глобальной переменной
window.myGlobalVar = "Я глобальная переменная";
console.log(myGlobalVar); // "Я глобальная переменная"
```

### Объект `global` в Node.js

В Node.js глобальный объект называется `global`:

```javascript
// В Node.js
console.log(global); // объект global
console.log(this === global); // false (в модулях)

// Добавление глобальной переменной
global.myGlobalVar = "Я глобальная переменная в Node.js";
console.log(myGlobalVar); // "Я глобальная переменная в Node.js"
```

## Объект `console`

Объект `console` доступен как в браузере, так и в Node.js и предоставляет методы для вывода информации.

### Основные методы console

#### `console.log()`

Основной метод для вывода информации:

```javascript
// Простой вывод
console.log("Привет, мир!");

// Вывод переменных
const name = "Иван";
const age = 25;
console.log("Имя:", name, "Возраст:", age);

// Шаблонные строки
console.log(`Привет, ${name}! Тебе ${age} лет.`);

// Вывод объектов
const user = { name: "Мария", age: 30 };
console.log("Пользователь:", user);

// Вывод массивов
const numbers = [1, 2, 3, 4, 5];
console.log("Числа:", numbers);
```

#### `console.error()`

Вывод сообщений об ошибках (обычно красным цветом):

```javascript
console.error("Произошла ошибка!");
console.error("Ошибка:", new Error("Что-то пошло не так"));

// В браузере выводится в консоль ошибок
// В Node.js выводится в stderr
```

#### `console.warn()`

Вывод предупреждений (обычно желтым цветом):

```javascript
console.warn("Внимание: устаревший метод");
console.warn("Предупреждение:", "Низкий уровень заряда батареи");
```

#### `console.info()`

Информационные сообщения (обычно синим цветом):

```javascript
console.info("Информация: сервер запущен");
console.info("Версия приложения:", "1.0.0");
```

### Дополнительные методы console

#### `console.table()`

Вывод данных в виде таблицы:

```javascript
// Массив объектов
const users = [
  { id: 1, name: "Иван", age: 25, city: "Москва" },
  { id: 2, name: "Мария", age: 30, city: "СПб" },
  { id: 3, name: "Петр", age: 28, city: "Казань" },
];

console.table(users);

// Объект
const person = {
  name: "Анна",
  age: 32,
  profession: "Разработчик",
  experience: 5,
};

console.table(person);

// Указание конкретных колонок
console.table(users, ["name", "age"]);
```

#### `console.group()` и `console.groupEnd()`

Группировка сообщений:

```javascript
console.group("Информация о пользователе");
console.log("Имя: Иван");
console.log("Возраст: 25");
console.log("Email: ivan@example.com");
console.groupEnd();

console.log("Активных: 120");
console.log("Заблокированных: 5");
console.groupEnd();

// Сворачиваемые группы
console.groupCollapsed("Детальная информация");
console.log("Детали...");
console.log("Больше деталей...");
console.groupEnd();
```

#### `console.time()` и `console.timeEnd()`

Измерение времени выполнения:

```javascript
// Простое измерение времени
console.time("Обработка данных");
// ... код для измерения
setTimeout(() => {
  console.timeEnd("Обработка данных"); // Обработка данных: 1005.123ms
}, 1000);

// Несколько таймеров
console.time("timer1");
console.time("timer2");

setTimeout(() => {
  console.timeEnd("timer1");
}, 500);

setTimeout(() => {
  console.timeEnd("timer2");
}, 1000);
```

#### `console.count()` и `console.countReset()`

Подсчет вызовов:

```javascript
function processItem(item) {
  console.count("Обработано элементов");
  // обработка элемента
}

processItem("A"); // Обработано элементов: 1
processItem("B"); // Обработано элементов: 2
processItem("C"); // Обработано элементов: 3

// Сброс счетчика
console.countReset("Обработано элементов");
processItem("D"); // Обработано элементов: 1

// Именованные счетчики
console.count("ошибки");
console.count("успех");
console.count("ошибки");
console.count("успех");
// ошибки: 1
// успех: 1
// ошибки: 2
// успех: 2
```

#### `console.trace()`

Вывод стека вызовов:

```javascript
function firstFunction() {
  secondFunction();
}

function secondFunction() {
  thirdFunction();
}

function thirdFunction() {
  console.trace("Трассировка стека");
}

firstFunction();
// Выведет полный стек вызовов
```

#### `console.assert()`

Условный вывод (только если условие ложно):

```javascript
const x = 5;
const y = 10;

console.assert(x > y, "x должно быть больше y");
// Выведет ошибку, так как 5 не больше 10

console.assert(x < y, "x должно быть меньше y");
// Ничего не выведет, так как условие истинно

// С дополнительными данными
const user = { name: "Иван", age: 17 };
console.assert(
  user.age >= 18,
  "Пользователь должен быть совершеннолетним",
  user
);
```

#### `console.clear()`

Очистка консоли:

```javascript
console.log("Это сообщение будет удалено");
console.clear(); // Очистка консоли
console.log("Это сообщение останется");
```

### Форматирование вывода

#### Подстановочные знаки

```javascript
// %s - строка
console.log("Привет, %s!", "Мир");

// %d или %i - целое число
console.log("Возраст: %d лет", 25);

// %f - число с плавающей точкой
console.log("Цена: %f рублей", 199.99);

// %o - объект
console.log("Пользователь: %o", { name: "Иван", age: 25 });

// %c - CSS стили (только в браузере)
console.log(
  "%cЭто красивый текст",
  "color: red; font-size: 20px; font-weight: bold;"
);
```

#### Стилизация в браузере

```javascript
// Цветной текст
console.log("%cУспех!", "color: green; font-weight: bold;");
console.log("%cОшибка!", "color: red; font-weight: bold;");
console.log("%cПредупреждение!", "color: orange; font-weight: bold;");

// Комбинирование стилей
console.log(
  "%cПроект%c готов к %cзапуску%c!",
  "color: blue; font-weight: bold;",
  "color: black;",
  "color: green; font-weight: bold;",
  "color: black;"
);

// Фоновые цвета
console.log("%c ВНИМАНИЕ ", "background: red; color: white; padding: 2px;");
```

## Модальные окна в браузере

### `alert()`

Показывает сообщение пользователю:

```javascript
alert("Добро пожаловать на сайт!");
alert("Операция выполнена успешно");

// С переменными
const userName = "Иван";
alert(`Привет, ${userName}!`);
```

### `confirm()`

Показывает диалог с кнопками "OK" и "Отмена":

```javascript
const result = confirm("Вы уверены, что хотите удалить файл?");

if (result) {
  console.log("Пользователь подтвердил действие");
  // код удаления файла
} else {
  console.log("Пользователь отменил действие");
}

// Практический пример
function deleteUser(userId) {
  const confirmed = confirm(
    "Удалить пользователя? Это действие нельзя отменить."
  );

  if (confirmed) {
    // Логика удаления
    console.log(`Пользователь ${userId} удален`);
    return true;
  }

  return false;
}
```

### `prompt()`

Запрашивает ввод от пользователя:

```javascript
// Простой запрос
const name = prompt("Как вас зовут?");
console.log(`Привет, ${name}!`);

// С значением по умолчанию
const age = prompt("Сколько вам лет?", "18");
console.log(`Вам ${age} лет`);

// Проверка ввода
const email = prompt("Введите ваш email:");

if (email === null) {
  console.log("Пользователь отменил ввод");
} else if (email === "") {
  console.log("Пользователь ничего не ввел");
} else {
  console.log(`Email: ${email}`);
}

// Валидация ввода
function getValidAge() {
  let age;

  do {
    age = prompt("Введите ваш возраст (18-100):");

    if (age === null) {
      return null; // Пользователь отменил
    }

    age = parseInt(age);

    if (isNaN(age) || age < 18 || age > 100) {
      alert("Неверный возраст. Попробуйте еще раз.");
      age = null;
    }
  } while (age === null);

  return age;
}

const validAge = getValidAge();
if (validAge) {
  console.log(`Возраст: ${validAge}`);
}
```

## Практические примеры

### Система логирования

```javascript
class Logger {
  constructor(level = "info") {
    this.level = level;
    this.levels = {
      error: 0,
      warn: 1,
      info: 2,
      debug: 3,
    };
  }

  log(level, message, ...args) {
    if (this.levels[level] <= this.levels[this.level]) {
      const timestamp = new Date().toISOString();
      const prefix = `[${timestamp}] [${level.toUpperCase()}]`;

      switch (level) {
        case "error":
          console.error(prefix, message, ...args);
          break;
        case "warn":
          console.warn(prefix, message, ...args);
          break;
        case "debug":
          console.debug(prefix, message, ...args);
          break;
        default:
          console.log(prefix, message, ...args);
      }
    }
  }

  error(message, ...args) {
    this.log("error", message, ...args);
  }

  warn(message, ...args) {
    this.log("warn", message, ...args);
  }

  info(message, ...args) {
    this.log("info", message, ...args);
  }

  debug(message, ...args) {
    this.log("debug", message, ...args);
  }
}

// Использование
const logger = new Logger("debug");

logger.info("Приложение запущено");
logger.warn("Низкий уровень памяти");
logger.error("Ошибка подключения к базе данных");
logger.debug("Детальная информация для отладки");
```

### Интерактивная консольная игра

```javascript
class GuessNumberGame {
  constructor() {
    this.secretNumber = Math.floor(Math.random() * 100) + 1;
    this.attempts = 0;
    this.maxAttempts = 7;
  }

  start() {
    console.log('%c🎮 ИГРА "УГАДАЙ ЧИСЛО" 🎮', "font-size: 20px; color: blue;");
    console.log("Я загадал число от 1 до 100. У вас есть 7 попыток!");

    this.playRound();
  }

  playRound() {
    if (this.attempts >= this.maxAttempts) {
      console.log(
        `💀 Игра окончена! Загаданное число было: ${this.secretNumber}`
      );
      this.askPlayAgain();
      return;
    }

    const guess = prompt(
      `Попытка ${this.attempts + 1}/${this.maxAttempts}. Введите число:`
    );

    if (guess === null) {
      console.log("Игра прервана пользователем");
      return;
    }

    const number = parseInt(guess);

    if (isNaN(number) || number < 1 || number > 100) {
      alert("Введите число от 1 до 100!");
      this.playRound();
      return;
    }

    this.attempts++;
    this.checkGuess(number);
  }

  checkGuess(guess) {
    console.log(`Ваш ответ: ${guess}`);

    if (guess === this.secretNumber) {
      console.log(`🎉 Поздравляю! Вы угадали за ${this.attempts} попыток!`);
      this.askPlayAgain();
    } else if (guess < this.secretNumber) {
      console.log("📈 Загаданное число больше");
      this.playRound();
    } else {
      console.log("📉 Загаданное число меньше");
      this.playRound();
    }
  }

  askPlayAgain() {
    const playAgain = confirm("Хотите сыграть еще раз?");

    if (playAgain) {
      this.secretNumber = Math.floor(Math.random() * 100) + 1;
      this.attempts = 0;
      this.start();
    } else {
      console.log("Спасибо за игру! 👋");
    }
  }
}
```

### Профилировщик производительности

```javascript
class Profiler {
  constructor() {
    this.timers = new Map();
    this.counters = new Map();
  }

  // Запуск таймера
  time(label) {
    this.timers.set(label, performance.now());
    console.log(`⏱️ Таймер "${label}" запущен`);
  }

  // Остановка таймера
  timeEnd(label) {
    const startTime = this.timers.get(label);

    if (startTime === undefined) {
      console.warn(`Таймер "${label}" не найден`);
      return;
    }

    const elapsed = performance.now() - startTime;
    this.timers.delete(label);

    console.log(`⏱️ Таймер "${label}": ${elapsed.toFixed(3)}ms`);
    return elapsed;
  }

  // Счетчик вызовов
  count(label) {
    const current = this.counters.get(label) || 0;
    const newCount = current + 1;
    this.counters.set(label, newCount);

    console.log(`🔢 ${label}: ${newCount}`);
    return newCount;
  }

  // Сброс счетчика
  countReset(label) {
    this.counters.set(label, 0);
    console.log(`🔄 Счетчик "${label}" сброшен`);
  }

  // Отчет по всем метрикам
  report() {
    console.group("📊 Отчет профилировщика");

    console.log("Активные таймеры:");
    this.timers.forEach((startTime, label) => {
      const elapsed = performance.now() - startTime;
      console.log(`  ${label}: ${elapsed.toFixed(3)}ms (активен)`);
    });

    console.log("Счетчики:");
    this.counters.forEach((count, label) => {
      console.log(`  ${label}: ${count}`);
    });

    console.groupEnd();
  }
}

// Использование
const profiler = new Profiler();

profiler.time("data-processing");

// Симуляция обработки данных
for (let i = 0; i < 1000000; i++) {
  profiler.count("iterations");
}

profiler.timeEnd("data-processing");
profiler.report();
```

### Группировка логов по категориям

```javascript
class CategoryLogger {
  constructor(categories = ["info", "error", "debug"]) {
    this.enabledCategories = new Set(categories);
  }

  log(category, message, ...args) {
    if (this.enabledCategories.has(category)) {
      console.group(`[${category.toUpperCase()}]`);
      console.log(message, ...args);
      console.groupEnd();
    }
  }

  enable(category) {
    this.enabledCategories.add(category);
  }

  disable(category) {
    this.enabledCategories.delete(category);
  }
}

const logger = new CategoryLogger(["api", "user"]);

logger.log("api", "Запрос к API", { url: "/users", method: "GET" });
logger.log("user", "Пользователь вошел в систему", { id: 123 });
logger.log("debug", "Детальная отладка", "не будет выведено");
```
